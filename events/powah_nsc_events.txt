country_event = {
	id = powah_nsc.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = has_nsc_active
		NOT = { has_technology = "tech_graviton_power_fake" }
		has_technology = "tech_graviton_power"
	}

	immediate = {
		give_technology = {
			tech = "tech_graviton_power_fake"
			message = no
		}
	}
}
